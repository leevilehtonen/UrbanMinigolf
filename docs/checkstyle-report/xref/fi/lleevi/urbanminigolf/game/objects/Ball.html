1   package fi.lleevi.urbanminigolf.game.objects;
2   
3   import java.awt.Color;
4   import java.awt.Graphics2D;
5   
6   public class Ball extends MovableGameObject {
7   
8       public static final int BALL_SIZE = 10;
9       public static final double BALL_FRICTION = 0.985;
10      public static final double BALL_FRICTION_THRESHOLD = 3;
11      public static final double BALL_HIT_MULTIPLIER = 2;
12  
13      private boolean hittable = true;
14      private boolean inHole = false;
15  
16      public Ball(double posX, double posY) {
17          super(posX, posY, BALL_SIZE, BALL_SIZE, Type.Ball);
18      }
19  
20      @Override
21      public void update(double delta) {
22  
23          setPosX(getPosX() + delta * getVelX());
24          setPosY(getPosY() + delta * getVelY());
25  
26          if (Math.abs(getVelX()) > BALL_FRICTION_THRESHOLD) {
27              setVelX(getVelX() * BALL_FRICTION);
28          } else {
29              setVelX(0);
30          }
31          if (Math.abs(getVelY()) > BALL_FRICTION_THRESHOLD) {
32              setVelY(getVelY() * BALL_FRICTION);
33          } else {
34              setVelY(0);
35          }
36  
37          if (Math.abs(getVelX()) < BALL_FRICTION_THRESHOLD && Math.abs(getVelY()) < BALL_FRICTION_THRESHOLD) {
38              hittable = true;
39          }
40      }
41  
42      @Override
43      public void render(Graphics2D g) {
44          if (!inHole) {
45              g.setColor(Color.WHITE);
46              g.fillOval(getBounds().x, getBounds().y, BALL_SIZE, BALL_SIZE);
47              g.setColor(Color.BLACK);
48              g.drawOval(getBounds().x, getBounds().y, BALL_SIZE, BALL_SIZE);
49          }
50      }
51  
52      public void hit(double velX, double velY) {
53          setVelX(velX * BALL_HIT_MULTIPLIER);
54          setVelY(velY * BALL_HIT_MULTIPLIER);
55      }
56  
57      public void intersectsWith(GameObject object) {
58          if (object.getType() == Type.Hole) {
59              Hole hole = (Hole) object;
60              if (hole.intersectsBall(this)) {
61                  inHole = true;
62              }
63          }
64          if (object.getType() == Type.Wall) {
65              Wall wall = (Wall) object;
66              if (wall.hitBottom(this) || wall.hitTop(this)) {
67                  mirrorYVel();
68              }
69              if (wall.hitLeft(this) || wall.hitRight(this)) {
70                  mirrorXVel();
71              }
72          }
73      }
74  
75      public boolean isHittable() {
76          return hittable;
77      }
78  
79      public void setHittable(boolean hittable) {
80          this.hittable = hittable;
81      }
82  
83      public boolean isInHole() {
84          return inHole;
85      }
86  
87      public void setInHole(boolean inHole) {
88          this.inHole = inHole;
89      }
90  
91  }
